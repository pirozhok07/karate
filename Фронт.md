```typescript
import axios from 'axios';
import { Tournament, Participant, Judge, Bracket, Match, Registration } from '../types';

const API_BASE_URL = 'http://localhost:8000/api';

const api = axios.create({
  baseURL: API_BASE_URL,
  headers: {
    'Content-Type': 'application/json',
  },
});

// Интерцептор для обработки ошибок
api.interceptors.response.use(
  (response) => response,
  (error) => {
    console.error('API Error:', error);
    return Promise.reject(error);
  }
);

export const tournamentAPI = {
  getAll: () => api.get<Tournament[]>('/tournaments/tournaments/'),
  getById: (id: number) => api.get<Tournament>(`/tournaments/tournaments/${id}/`),
  create: (data: Partial<Tournament>) => api.post<Tournament>('/tournaments/tournaments/', data),
  update: (id: number, data: Partial<Tournament>) => api.put<Tournament>(`/tournaments/tournaments/${id}/`, data),
  delete: (id: number) => api.delete(`/tournaments/tournaments/${id}/`),
};

export const participantAPI = {
  getAll: () => api.get<Participant[]>('/participants/participants/'),
  getById: (id: number) => api.get<Participant>(`/participants/participants/${id}/`),
  create: (data: Partial<Participant>) => api.post<Participant>('/participants/participants/', data),
  update: (id: number, data: Partial<Participant>) => api.put<Participant>(`/participants/participants/${id}/`, data),
  delete: (id: number) => api.delete(`/participants/participants/${id}/`),
};

export const registrationAPI = {
  getAll: () => api.get<Registration[]>('/participants/registrations/'),
  getById: (id: number) => api.get<Registration>(`/participants/registrations/${id}/`),
  create: (data: Partial<Registration>) => api.post<Registration>('/participants/registrations/', data),
  update: (id: number, data: Partial<Registration>) => api.put<Registration>(`/participants/registrations/${id}/`, data),
  approve: (id: number) => api.post(`/participants/registrations/${id}/approve/`),
  reject: (id: number) => api.post(`/participants/registrations/${id}/reject/`),
};

export const judgeAPI = {
  getAll: () => api.get<Judge[]>('/judges/judges/'),
  getById: (id: number) => api.get<Judge>(`/judges/judges/${id}/`),
  create: (data: Partial<Judge>) => api.post<Judge>('/judges/judges/', data),
  update: (id: number, data: Partial<Judge>) => api.put<Judge>(`/judges/judges/${id}/`, data),
  delete: (id: number) => api.delete(`/judges/judges/${id}/`),
};

export const bracketAPI = {
  getAll: () => api.get<Bracket[]>('/brackets/brackets/'),
  getById: (id: number) => api.get<Bracket>(`/brackets/brackets/${id}/`),
  create: (data: Partial<Bracket>) => api.post<Bracket>('/brackets/brackets/', data),
  update: (id: number, data: Partial<Bracket>) => api.put<Bracket>(`/brackets/brackets/${id}/`, data),
  generate: (id: number) => api.post(`/brackets/brackets/${id}/generate/`),
  start: (id: number) => api.post(`/brackets/brackets/${id}/start/`),
};

export const matchAPI = {
  getAll: () => api.get<Match[]>('/brackets/matches/'),
  getById: (id: number) => api.get<Match>(`/brackets/matches/${id}/`),
  setWinner: (id: number, winnerId: number) => api.post(`/brackets/matches/${id}/set_winner/`, { winner_id: winnerId }),
};

export default api;
```

```typescript
import React, { useState, useEffect } from 'react';
import {
  Grid,
  Card,
  CardContent,
  Typography,
  Box,
  CircularProgress,
} from '@mui/material';
import {
  Tournament as TournamentIcon,
  People as PeopleIcon,
  Gavel as GavelIcon,
  EmojiEvents as TrophyIcon,
} from '@mui/icons-material';
import { tournamentAPI, participantAPI, judgeAPI, bracketAPI } from '../services/api';

const Dashboard: React.FC = () => {
  const [stats, setStats] = useState({
    tournaments: 0,
    participants: 0,
    judges: 0,
    brackets: 0,
  });
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const fetchStats = async () => {
      try {
        const [tournamentsRes, participantsRes, judgesRes, bracketsRes] = await Promise.all([
          tournamentAPI.getAll(),
          participantAPI.getAll(),
          judgeAPI.getAll(),
          bracketAPI.getAll(),
        ]);

        setStats({
          tournaments: tournamentsRes.data.length,
          participants: participantsRes.data.length,
          judges: judgesRes.data.length,
          brackets: bracketsRes.data.length,
        });
      } catch (error) {
        console.error('Error fetching stats:', error);
      } finally {
        setLoading(false);
      }
    };

    fetchStats();
  }, []);

  if (loading) {
    return (
      <Box display="flex" justifyContent="center" alignItems="center" minHeight="50vh">
        <CircularProgress />
      </Box>
    );
  }

  const statCards = [
    { title: 'Турниры', value: stats.tournaments, icon: <TournamentIcon />, color: '#1976d2' },
    { title: 'Участники', value: stats.participants, icon: <PeopleIcon />, color: '#2e7d32' },
    { title: 'Судьи', value: stats.judges, icon: <GavelIcon />, color: '#ed6c02' },
    { title: 'Турнирные сетки', value: stats.brackets, icon: <TrophyIcon />, color: '#9c27b0' },
  ];

  return (
    <Box>
      <Typography variant="h4" gutterBottom>
        Дашборд
      </Typography>
      <Typography variant="subtitle1" color="text.secondary" gutterBottom>
        Обзор системы проведения соревнований
      </Typography>
      
      <Grid container spacing={3} sx={{ mt: 2 }}>
        {statCards.map((stat) => (
          <Grid item xs={12} sm={6} md={3} key={stat.title}>
            <Card>
              <CardContent>
                <Box display="flex" alignItems="center" justifyContent="space-between">
                  <Box>
                    <Typography color="text.secondary" gutterBottom>
                      {stat.title}
                    </Typography>
                    <Typography variant="h4" component="div">
                      {stat.value}
                    </Typography>
                  </Box>

                    {stat.icon}
```

```typescript
import React, { useState, useEffect } from 'react';
import {
  Box,
  Typography,
  Button,
  Paper,
  Table,
  TableBody,
  TableCell,
  TableContainer,
  TableHead,
  TableRow,
  Chip,
  IconButton,
  Dialog,
  DialogTitle,
  DialogContent,
  DialogActions,
  TextField,
  MenuItem,
} from '@mui/material';
import { Add as AddIcon, Edit as EditIcon } from '@mui/icons-material';
import { Tournament } from '../types';
import { tournamentAPI } from '../services/api';

const TournamentsPage: React.FC = () => {
  const [tournaments, setTournaments] = useState<Tournament[]>([]);
  const [loading, setLoading] = useState(true);
  const [openDialog, setOpenDialog] = useState(false);
  const [editingTournament, setEditingTournament] = useState<Tournament | null>(null);

  useEffect(() => {
    fetchTournaments();
  }, []);

  const fetchTournaments = async () => {
    try {
      const response = await tournamentAPI.getAll();
      setTournaments(response.data);
    } catch (error) {
      console.error('Error fetching tournaments:', error);
    } finally {
      setLoading(false);
    }
  };

  const handleCreate = () => {
    setEditingTournament(null);
    setOpenDialog(true);
  };

  const handleEdit = (tournament: Tournament) => {
    setEditingTournament(tournament);
    setOpenDialog(true);
  };

  const handleSave = async (tournamentData: Partial<Tournament>) => {
    try {
      if (editingTournament) {
        await tournamentAPI.update(editingTournament.id, tournamentData);
      } else {
        await tournamentAPI.create(tournamentData);
      }
      setOpenDialog(false);
      fetchTournaments();
    } catch (error) {
      console.error('Error saving tournament:', error);
    }
  };

  const getStatusColor = (status: string) => {
    switch (status) {
      case 'upcoming': return 'primary';
      case 'ongoing': return 'secondary';
      case 'completed': return 'success';
      case 'cancelled': return 'error';
      default: return 'default';
    }
  };

  const getTypeText = (type: string) => {
    return type === 'kumite' ? 'Кумите' : 'Ката';
  };

  return (
    <Box>
      <Box display="flex" justifyContent="space-between" alignItems="center" mb={3}>
        <Typography variant="h4">Турниры</Typography>
        <Button
          variant="contained"
          startIcon={<AddIcon />}
          onClick={handleCreate}
        >
          Создать турнир
        </Button>
      </Box>

      <TableContainer component={Paper}>
        <Table>
          <TableHead>
            <TableRow>
              <TableCell>Название</TableCell>
              <TableCell>Тип</TableCell>
              <TableCell>Дата начала</TableCell>
              <TableCell>Место проведения</TableCell>
              <TableCell>Статус</TableCell>
              <TableCell>Действия</TableCell>
            </TableRow>
          </TableHead>
          <TableBody>
            {tournaments.map((tournament) => (
              <TableRow key={tournament.id}>
                <TableCell>{tournament.name}</TableCell>
                <TableCell>{getTypeText(tournament.tournament_type)}</TableCell>
                <TableCell>{new Date(tournament.start_date).toLocaleDateString()}</TableCell>
                <TableCell>{tournament.location}</TableCell>
                <TableCell>
                  <Chip
                    label={tournament.status}
                    color={getStatusColor(tournament.status) as any}
                    size="small"
                  />
                </TableCell>
                <TableCell>
                  <IconButton onClick={() => handleEdit(tournament)}>
                    <EditIcon />
                  </IconButton>
                </TableCell>
              </TableRow>
            ))}
          </TableBody>
        </Table>
      </TableContainer>

      <TournamentDialog
        open={openDialog}
        tournament={editingTournament}
        onClose={() => setOpenDialog(false)}
        onSave={handleSave}
      />
    </Box>
  );
};

// Компонент диалога для создания/редактирования турнира
interface TournamentDialogProps {
  open: boolean;
  tournament: Tournament | null;
  onClose: () => void;
  onSave: (data: Partial<Tournament>) => void;
}

const TournamentDialog: React.FC<TournamentDialogProps> = ({
  open,
  tournament,
  onClose,
  onSave,
}) => {
  const [formData, setFormData] = useState({
    name: '',
    description: '',
    tournament_type: 'kumite',
    start_date: '',
    end_date: '',
    location: '',
    status: 'upcoming',
    max_participants: 0,
  });

  useEffect(() => {
    if (tournament) {
      setFormData({
        name: tournament.name,
        description: tournament.description,
        tournament_type: tournament.tournament_type,
        start_date: tournament.start_date,
        end_date: tournament.end_date,
        location: tournament.location,
        status: tournament.status,
        max_participants: tournament.max_participants,
      });
    } else {
      setFormData({
        name: '',
        description: '',
        tournament_type: 'kumite',
        start_date: '',
        end_date: '',
        location: '',
        status: 'upcoming',
        max_participants: 0,
      });
    }
  }, [tournament]);

  const handleSubmit = () => {
    onSave(formData);
  };

  return (
    <Dialog open={open} onClose={onClose} maxWidth="md" fullWidth>
      <DialogTitle>
        {tournament ? 'Редактировать турнир' : 'Создать турнир'}
      </DialogTitle>
      <DialogContent>
        <Box display="flex" flexDirection="column" gap={2} mt={1}>
          <TextField
            label="Название турнира"
            value={formData.name}
            onChange={(e) => setFormData({ ...formData, name: e.target.value })}
            fullWidth
          />
          <TextField
            label="Описание"
            value={formData.description}
            onChange={(e) => setFormData({ ...formData, description: e.target.value })}
            multiline
            rows={3}
            fullWidth
          />
          <TextField
            select
            label="Тип турнира"
            value={formData.tournament_type}
            onChange={(e) => setFormData({ ...formData, tournament_type: e.target.value })}
            fullWidth
          >
            <MenuItem value="kumite">Кумите</MenuItem>
            <MenuItem value="kata">Ката</MenuItem>
          </TextField>
          <TextField
            label="Дата начала"
            type="datetime-local"
            value={formData.start_date}
            onChange={(e) => setFormData({ ...formData, start_date: e.target.value })}
            InputLabelProps={{ shrink: true }}
            fullWidth
          />
          <TextField
            label="Дата окончания"
            type="datetime-local"
            value={formData.end_date}
            onChange={(e) => setFormData({ ...formData, end_date: e.target.value })}
            InputLabelProps={{ shrink: true }}
            fullWidth
          />
          <TextField
            label="Место проведения"
            value={formData.location}
            onChange={(e) => setFormData({ ...formData, location: e.target.value })}
            fullWidth
          />
          <TextField
            select
            label="Статус"
            value={formData.status}
            onChange={(e) => setFormData({ ...formData, status: e.target.value })}
            fullWidth
          >
            <MenuItem value="upcoming">Предстоящий</MenuItem>
            <MenuItem value="ongoing">В процессе</MenuItem>
            <MenuItem value="completed">Завершен</MenuItem>
            <MenuItem value="cancelled">Отменен</MenuItem>
          </TextField>
          <TextField
            label="Максимальное количество участников"
            type="number"
            value={formData.max_participants}
            onChange={(e) => setFormData({ ...formData, max_participants: parseInt(e.target.value) })}
            fullWidth
          />
        </Box>
      </DialogContent>
      <DialogActions>
        <Button onClick={onClose}>Отмена</Button>
        <Button onClick={handleSubmit} variant="contained">
          {tournament ? 'Обновить' : 'Создать'}
        </Button>
      </DialogActions>
    </Dialog>
  );
};

export default TournamentsPage;
```
